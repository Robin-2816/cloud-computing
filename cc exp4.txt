import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
url = 'https://raw.githubusercontent.com/Robin-2816/cloud-computing/blob/main/student_info.csv'

# Load the dataset
df = pd.read_csv(url)

# Print the data
# Because there are a lot of data, use head() to only print the first few rows
df.head()


# Select X and y
X = df['study_hours'].values.reshape(-1, 1)  # Independent variable
y = df['student_marks'].values               # Dependent variable

# Select the Model
model = LinearRegression()


# Train the Model
model.fit(X, y)

# Predict the values
y_pred = model.predict(X)

# Print Slope and Intercept
print("Slope (m):", model.coef_[0])
print("Intercept (c):", model.intercept_)



# Plot the data
plt.scatter(X, y, color='blue', label='Actual Data')
plt.plot(X, y_pred, color='red', label='Regression Line')
plt.xlabel('Hours Studied')
plt.ylabel('Marks Obtained')
plt.title('Simple Linear Regression on Student Data')
plt.legend()
plt.grid()
plt.show()
